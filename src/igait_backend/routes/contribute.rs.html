27/// Takes in the `Multipart` request and unpacks the arguments into a `ContributeRequestArguments` object.
28/// 
29/// # Fails
30/// If any of the fields are missing or if the files are too large.
31/// 
32/// # Arguments
33/// * `multipart` - The `Multipart` object to unpack.
34#[tracing::instrument]
35async fn unpack_contribute_arguments(
36    multipart:   &mut Multipart
37) -> Result<ContributeRequestArguments> {
38    // Initialize all of the fields as options
39    let mut uid_option:       Option<String> = None;
40    let mut name_option:      Option<String> = None;
41    let mut email_option:     Option<String> = None;
42
43    // Initialize the file fields as options
44    let mut front_file_name_option:  Option<String> = None;
45    let mut side_file_name_option:   Option<String> = None;
46    let mut front_file_bytes_option: Option<Bytes>  = None;
47    let mut side_file_bytes_option:  Option<Bytes>  = None;
48
49    // Loop through the fields
50    while let Some(field) = multipart
51        .next_field().await
52        .context("Bad upload request! Is it possible you submitted a file over the size limit?")?
53    {
54        let name = field.name();
55        let field_name = field.file_name();
56        info!("Field Incoming: {name:?} - File Attached: {field_name:?}");
57        
58        match field.name() {
59            Some("fileuploadfront") => {
60                front_file_name_option = field
61                    .file_name().map(|x| String::from(x));
62                front_file_bytes_option = Some(field.bytes()
63                    .await
64                    .context("Could not unpack bytes from field 'fileuploadfront'! Was there no file attached?")?);
65            },
66            Some("fileuploadside") => {
67                side_file_name_option = field
68                    .file_name().map(|x| String::from(x));
69                side_file_bytes_option = Some(field.bytes()
70                    .await
71                    .context("Could not unpack bytes from field 'fileuploadside'! Was there no file attached?")?);
72            },
73            Some("email") => {
74                email_option = Some(
75                    field
76                        .text().await
77                        .context("Field 'uid' wasn't readable as text!")?
78                        .to_string());
79            }
80            Some("name") => {
81                name_option = Some(
82                    field
83                        .text().await
84                        .context("Field 'name' wasn't readable as text!")?
85                        .to_string());
86            }
87            Some("uid") => {
88                uid_option = Some(
89                    field
90                        .text().await
91                        .context("Field 'uid' wasn't readable as text!")?
92                        .to_string());
93            }
94            _ => {
95                info!("Which had an unknown/no field name...");
96            }
97        }
98    }
99
100    // Make sure all of the fields are present
101    let uid:   String = uid_option.ok_or(   anyhow!( "Missing 'uid' in request"   ))?;
102    let name:  String = name_option.ok_or( anyhow!( "Missing 'name' in request" ))?;
103    let email: String = email_option.ok_or( anyhow!( "Missing 'email' in request" ))?;
104
105    // Make sure all of the file fields are present
106    let front_file_name:  String = front_file_name_option.ok_or(  anyhow!( "Missing 'fileuploadfront' in request!" ))?;
107    let side_file_name:   String = side_file_name_option.ok_or(   anyhow!( "Missing 'fileuploadside' in request!"  ))?;
108    let front_file_bytes: Bytes  = front_file_bytes_option.ok_or( anyhow!( "Missing 'fileuploadfront' in request!" ))?;
109    let side_file_bytes:  Bytes  = side_file_bytes_option.ok_or(  anyhow!( "Missing 'fileuploadside' in request!"  ))?;
110
111    Ok(ContributeRequestArguments {
112        uid,
113        name,
114        email, 
115        front_file: ContributeRequestFile {
116            name: front_file_name, 
117            bytes: front_file_bytes
118        },
119        side_file: ContributeRequestFile {
120            name: side_file_name,
121            bytes: side_file_bytes
122        }
123    })
124}

126/// The entrypoint for the contribute request.
127/// 
128/// # Fails
129/// * If the arguments are missing.
130/// * If the files are too large.
131/// * If the files fail to save to S3.
132/// * If the job fails to save to the database.
133/// * If the welcome email fails to send.
134/// 
135/// # Arguments
136/// * `app` - The application state.
137/// * `multipart` - The `Multipart` object to unpack.
138#[tracing::instrument]
139pub async fn contribute_entrypoint(
140    State(app): State<AppStatePtr>,
141    mut multipart: Multipart
142) -> Result<(), AppError> {
143    let app = app.state;
144
145    info!("Unpacking arguments...");
146    // Unpack the arguments
147    let arguments: ContributeRequestArguments = unpack_contribute_arguments(
148            &mut multipart
149        ).await
150        .context("Failed to unpack arguments!")?;
151
152    // Try to save the files to S3
153    if let Err(err) = 
154        save_upload_files( 
155            app.clone(),
156            arguments.front_file,
157            arguments.side_file,
158            &arguments.uid,
159            &arguments.email,
160            &arguments.name
161        ).await 
162    {
163        return Err(AppError(err
164            .context("Failed to save locally or upload files to S3!")));
165    }
166
167    // Thank the user for their contribution
168    send_contribution_email(
169        app.clone(),
170        &arguments.email,
171        &arguments.name
172    )
173        .await
174        .context("Failed to send contribution email!")?;
175    info!("Successfully sent contribution email!");
176
177    Ok(())
178}

180/// Saves the upload files to S3 and the local filesystem.
181/// 
182/// # Fails
183/// * If the files fail to save to S3.
184/// * If the files fail to save to the local filesystem.
185/// 
186/// # Arguments
187/// * `app` - The application state.
188/// * `front_file` - The front file to save.
189/// * `side_file` - The side file to save.
190/// * `user_id` - The user ID to save the files under.
191/// * `email` - The email to save the files under.
192/// * `name` - The name to save the files under.
193#[tracing::instrument]
194async fn save_upload_files<'a> (
195    app:              Arc<AppState>,
196    front_file:       ContributeRequestFile,
197    side_file:        ContributeRequestFile,
198    user_id:          &str,
199    email:            &str,
200    name:             &str,
201) -> Result<()> {
202    // Unpack the extensions
203    let front_extension = front_file.name.split('.')
204        .last()
205        .context("Must have a file extension!")?;
206    let side_extension = side_file.name.split('.')
207        .last()
208        .context("Must have a file extension!")?;
209    
210    // Ensure a directory exists for this file ID
211    let unix_timestamp = SystemTime::now()
212        .duration_since(SystemTime::UNIX_EPOCH)
213        .context("Failed to get the current time!")?
214        .as_secs();
215    let email_user_id = format!("{};{}", user_id, email.replace('@', "_at_"));
216
217    // Build byte vectors
218    let mut front_byte_vec: Vec<u8> = Vec::new();
219    let mut side_byte_vec: Vec<u8> = Vec::new();
220    front_byte_vec.write_all(&front_file.bytes)
221        .await
222        .context("Failed to build u8 vector from the front file's Bytes object!")?;
223    side_byte_vec.write_all(&side_file.bytes)
224        .await
225        .context("Failed to build u8 vector from side file's Bytes object!")?;
226
227    // Upload the all three files to S3
228    app.bucket
229        .lock().await
230        .put_object(format!("research/{}/{}/front.{}", email_user_id, unix_timestamp, front_extension), &front_byte_vec)
231        .await 
232        .context("Failed to upload front file to S3! Continuing regardless.")?;
233    info!("Successfully uploaded front file to S3!");
234    app.bucket
235        .lock().await
236        .put_object(format!("research/{}/{}/side.{}", email_user_id, unix_timestamp, side_extension), &side_byte_vec)
237        .await
238        .context("Failed to upload front side to S3! Continuing regardless.")?;
239    info!("Successfully uploaded side file to S3!");
240    info!("Successfully saved all files physically and to S3!");
241    
242    // Return as successful
243    Ok(())
244}
