22/// Takes in the `Multipart` request and unpacks the arguments into a `HistoricalRequestArguments` object.
23/// 
24/// # Fails
25/// If any of the fields are missing or if the files are too large.
26/// 
27/// # Arguments
28/// * `multipart` - The `Multipart` object to unpack.
29#[tracing::instrument]
30async fn unpack_historical_arguments(
31    mut multipart: Multipart
32) -> Result<HistoricalRequestArguments> {
33    // Create placeholders for all fields
34    let mut uid_option: Option<String> = None;
35    let mut entries_option: Option<usize> = None;
36    let mut result_type_option: Option<String> = None;
37    let mut date_range: (Option<SystemTime>, Option<SystemTime>) = (None, None);
38    let mut include_original_option = None;
39    let mut include_skeleton_option = None;
40
41    while let Some(field) = multipart
42        .next_field().await
43        .context("Bad request! Is it possible you submitted a file over the size limit?")?
44    {
45        let name = field.name();
46        info!("Field Incoming: {name:#?}");
47        match field.name() {
48            Some("user_id") => {
49                uid_option = Some(
50                    field
51                        .text().await
52                        .context("Field 'user_id' wasn't readable as text!")?);
53            },
54            Some("entries") => {
55                entries_option = Some(
56                    field
57                        .text().await
58                        .context("Field 'user_id' wasn't readable as text!")?
59                        .parse::<usize>()
60                        .context("Field 'entries' didn't contain a valid integer!")?);
61
62                    if let Some(entries) = &entries_option {
63                        if *entries == 0 {
64                            bail!("Field 'entries' must be greater than 0!");
65                        }
66                    }
67            },
68            Some("result_type") => {
69                result_type_option = Some(field
70                    .text().await
71                    .context("Field 'result_type' wasn't readable as text!")?);
72
73                if let Some(result_type) = &result_type_option {
74                    if result_type != "ASD" && result_type != "NO ASD" {
75                        bail!("Field 'result_type' must be either 'ASD' or 'NO ASD'!");
76                    }
77                }
78            },
79            Some("start_timestamp") => {
80                date_range.0 = Some(system_time_from_secs(
81                    serde_json::Value::Number(field
82                    .text().await
83                    .context("Field 'start_timestamp' wasn't readable as text!")?
84                    .parse::<u64>()
85                    .context("Couldn't parse field 'start_timestamp' into a 64-bit integer!")?
86                    .into()
87                )).context("Field 'start_timestamp' was an integer, but not a valid UNIX timestamp!")?);
88            },
89            Some("end_timestamp") => {
90                date_range.1 = Some(system_time_from_secs(
91                    serde_json::Value::Number(field
92                        .text().await
93                        .context("Field 'end_timestamp' wasn't readable as text!")?
94                        .parse::<u64>()
95                        .context("Couldn't parse field 'end_timestamp' into a 64-bit integer!")?
96                        .into()
97                )).context("Field 'end_timestamp' was an integer, but not a valid UNIX timestamp!")?);
98            },
99            Some("include_original") => {
100                include_original_option = Some(
101                    field
102                        .text().await
103                        .context("Field 'include_original' wasn't readable as text!")?
104                        .parse::<bool>()
105                        .context("Field 'include_original' didn't contain a valid boolean!")?);
106            },
107            Some("include_skeleton") => {
108                include_skeleton_option = Some(
109                    field
110                        .text().await
111                        .context("Field 'include_skeleton' wasn't readable as text!")?
112                        .parse::<bool>()
113                        .context("Field 'include_skeleton' didn't contain a valid boolean!")?);
114            },
115            other => {
116                warn!("Recieved unknown/no field name '{other:?}'!");
117            }
118        }
119    }
120
121    Ok(HistoricalRequestArguments {
122        uid: uid_option.ok_or(anyhow!("Missing 'user_id' in request!"))?,
123        entries: entries_option,
124        result_type: result_type_option,
125        date_range: date_range,
126        include_original: include_original_option.unwrap_or(false),
127        include_skeleton: include_skeleton_option.unwrap_or(false)
128    })
129}

131/// The entrypoint for the historical submissions endpoint.
132/// 
133/// # Fails
134/// * If the arguments cannot be unpacked.
135/// * If the jobs cannot be retrieved from the database.
136/// * If the email cannot be sent.
137/// 
138/// # Arguments
139/// * `app` - The application state.
140/// * `multipart` - The `Multipart` object containing the request.
141#[tracing::instrument]
142pub async fn historical_entrypoint ( 
143    State(app): State<AppStatePtr>,
144    multipart: Multipart
145) -> Result<(), AppError> {
146    let app = app.state;
147
148    // Unpack the arguments
149    info!("Unpacking arguments...");
150    let arguments = unpack_historical_arguments(multipart).await
151        .context("Failed to unpack historical arguments!")?;
152
153    // Get all jobs
154    let mut jobs: Vec<(usize, Job)> = app.db
155        .lock().await
156        .get_all_jobs(
157            &arguments.uid
158        )
159        .await
160        .context("Failed to get jobs!")?
161        .into_iter()
162        .enumerate()
163        .collect();
164
165    // Remove the template job
166    jobs.remove(0);
167
168    // Filter by result type and start/end date
169    jobs = jobs.into_iter()
170        .filter(|(_, job)| {
171            // Filter by result type
172            if let Some(result_type) = &arguments.result_type {
173                if result_type == "ASD" {
174                    if job.status.code != JobStatusCode::Complete ||
175                       job.status.value != "ASD"
176                    {
177                        return false;
178                    }
179                } else if result_type == "NO ASD" {
180                    if job.status.code != JobStatusCode::Complete || 
181                       job.status.value != "NO ASD"
182                    {
183                        return false;
184                    }
185                }
186            }
187
188            // Make sure it's after the start date
189            if let Some(start_date) = arguments.date_range.0 {
190                if job.timestamp < start_date {
191                    return false;
192                }
193            }
194
195            // Make sure it's before the end date
196            if let Some(end_date) = arguments.date_range.1 {
197                if job.timestamp > end_date {
198                    return false;
199                }
200            }
201
202            true
203        })
204        .collect();
205
206    jobs.reverse();
207
208    // Lastly, only the return the number of entries requested
209    if let Some(num_entries) = arguments.entries {
210        jobs.truncate(num_entries);
211    }
212    
213    // Generate the body
214    let mut email_body = concat!(
215        "<h1>Thank you for contacting iGait!</h1>",
216        "You recently requested a history of your submissions. Located below can be found, in chronological order, your past submissions.<br><br>",
217        "You selected the following filters:"
218    ).to_string();
219    email_body += &format!(
220        "<ul>\
221        <li>Entries: {}</li>\
222        <li>Result Type: {}</li>\
223        <li>Start Date: {}</li>\
224        <li>End Date: {}</li>\
225        <li>Include Original: {}</li>\
226        <li>Include Skeleton: {}</li>\
227        </ul><br>",
228        arguments.entries.map(|n| n.to_string()).unwrap_or(String::from("All")),
229        arguments.result_type.as_deref().unwrap_or("Any"),
230        arguments.date_range.0.map(|d| DateTime::<Utc>::from(d).with_timezone(&chrono_tz::US::Central).to_string()).unwrap_or("Any".to_string()),
231        arguments.date_range.1.map(|d| DateTime::<Utc>::from(d).with_timezone(&chrono_tz::US::Central).to_string()).unwrap_or("Any".to_string()),
232        if arguments.include_original { "Yes" } else { "No" },
233        if arguments.include_skeleton { "Yes" } else { "No" }
234    );
235
236    // Make sure that we have at least some results
237    if jobs.is_empty() {
238        return Err(AppError(anyhow!("There were no jobs that matched your query!")));
239    }
240
241    for (job_id, job) in jobs.iter() {
242        // Add a condensed version of the job to the shortened body
243        let dt_timestamp_utc: DateTime<Utc> = job.timestamp.into();
244        let dt_timestamp_cst = dt_timestamp_utc.with_timezone(&chrono_tz::US::Central);
245        email_body.push_str(&format!(
246            "<h2>{}</h2>",
247            dt_timestamp_cst.to_string()
248        ));
249        
250        match job.status.code {
251            JobStatusCode::Complete => {
252                email_body.push_str(
253                    &format!(
254                        "- Status: Complete<br>- Result: {}<br>", 
255                        job.status.value
256                    )
257                );
258                if arguments.include_skeleton {
259                    info!("Preparing to grab pre-signed URLs for the skeleton-overlaid videos...");
260
261                    let inputs_prefix = format!(
262                        "{}/outputs/{};{}/videos/",
263                        arguments.uid,
264                        arguments.uid, job_id
265                    );
266        
267                    // List the available files (since we don't know the extensions)
268                    let results = app
269                        .bucket
270                        .lock().await
271                        .list(inputs_prefix, Some("/".to_string())).await?;
272                    let mut front_file_key = None;
273                    let mut side_file_key = None;
274                    for object in &results[0].contents {
275                        let file = object.key
276                            .split("/")
277                            .last()
278                            .context("Missing filename! (Likely unreachable?)")?;
279                        let file_name = file.split(".")
280                            .next()
281                            .context("Empty filename! (Likely unreachable?)")?;
282                        info!("Found file `{file}` with name `{file_name}`");
283        
284                        if file_name.contains("__F_") {
285                            front_file_key = Some(object.key.clone());
286                        } else if file_name.contains("__S_") {
287                            side_file_key = Some(object.key.clone());
288                        }
289                    }
290        
291                    let front_skeleton_video_link = if let Some(front_file_key) = front_file_key {
292                        format!("<a href=\"{}\">Skeleton Front Video</a>",
293                            app
294                                .bucket
295                                .lock().await
296                                .presign_get(front_file_key, 86400 * 7, None)
297                                .unwrap_or(String::from("Failed to get the front skeleton keyframed URL!"))
298                        )
299                    } else {
300                        String::from("No front skeleton video available.")
301                    };
302                    let side_skeleton_video_link = if let Some(side_file_key) = side_file_key {
303                        format!("<a href=\"{}\">Skeleton Side Video</a>",
304                            app
305                                .bucket
306                                .lock().await
307                                .presign_get(side_file_key, 86400 * 7, None)
308                                .unwrap_or(String::from("Failed to get the side skeleton keyframed URL!"))
309                        )
310                    } else {
311                        String::from("No side skeleton video available.")
312                    };
313                    
314        
315                    email_body.push_str(&format!(
316                        "- {}<br>- {}<br>",
317                        front_skeleton_video_link,
318                        side_skeleton_video_link
319                    ));
320
321                    info!("Grabbed pre-signed URLs for the skeleton-overlaid videos!");
322                }
323            },
324            _ => {
325                email_body.push_str(&format!(
326                    "- Status: {:?}<br>- Additional Information: {}<br>",
327                    job.status.code,
328                    job.status.value
329                ));
330
331                if arguments.include_skeleton {
332                    email_body.push_str("Since this job does not have a complete result, we are unable to provide a skeletonized video.<br>");
333                }
334            }
335        }
336        
337        // Add the original videos, if requested
338        if arguments.include_original {
339            info!("Preparing to grab pre-signed URLs for the original videos...");
340            let inputs_prefix = format!(
341                "{}/inputs/{};{}/",
342                arguments.uid,
343                arguments.uid, job_id
344            );
345
346            // List the available files (since we don't know the extensions)
347            let results = app
348                .bucket
349                .lock().await
350                .list(inputs_prefix, Some("/".to_string())).await?;
351            let mut front_file_key = None;
352            let mut side_file_key = None;
353            for object in &results[0].contents {
354                let file = object.key
355                    .split("/")
356                    .last()
357                    .context("Missing filename! (Likely unreachable?)")?;
358                let file_name = file.split(".")
359                    .next()
360                    .context("Empty filename! (Likely unreachable?)")?;
361                info!("Found file `{file}` with name `{file_name}`");
362
363                if file_name == "front" {
364                    front_file_key = Some(object.key.clone());
365                } else if file_name == "side" {
366                    side_file_key = Some(object.key.clone());
367                }
368            }
369
370            let front_original_video_link = if let Some(front_file_key) = front_file_key {
371                format!("<a href=\"{}\">Original Front Video</a>",
372                    app
373                        .bucket
374                        .lock().await
375                        .presign_get(front_file_key, 86400 * 7, None)
376                        .unwrap_or(String::from("Failed to get the front keyframed URL!"))
377                )
378            } else {
379                String::from("No front original video available.")
380            };
381            let side_original_video_link = if let Some(side_file_key) = side_file_key {
382                format!("<a href=\"{}\">Original Side Video</a>",
383                    app
384                        .bucket
385                        .lock().await
386                        .presign_get(side_file_key, 86400 * 7, None)
387                        .unwrap_or(String::from("Failed to get the side keyframed URL!"))
388                )
389            } else {
390                String::from("No side original video available.")
391            };
392
393            email_body.push_str(&format!(
394                "- {}<br>- {}<br>",
395                front_original_video_link,
396                side_original_video_link
397            ));
398
399            info!("Grabbed pre-signed URLs for the original videos!");
400        }
401    }
402    info!("Built the HTML file and email body!");
403
404    // Now that we have created the shortened body, let's 
405    //  upload the more verbose file to the user's S3,
406    //  and attach the link to the shortened body
407    let timestamp = SystemTime::now()
408        .duration_since(SystemTime::UNIX_EPOCH)
409        .context("Unreachable - We are not time travelers ^^`")?
410        .as_secs();
411
412    // Generate the PDF
413    let (email, pdf_link) = get_email_and_pdf_link(app.clone(), jobs, arguments.uid, timestamp)
414        .await
415        .context("Failed to generate the PDF file!")?;
416
417    
418    // Add the link to the email body
419    email_body += &format!("<br><br><h3>Complete Historical Data</h3><p>If you would like to see patient data for each submission, please click <a href=\"{}\">here</a></p><br>", pdf_link);
420    email_body += "<br><br><h2>Please contact &lt;contact email here&gt; with any additional questions!</h2>";
421
422    // Send the email to the email in the first job
423    //  (This is a bit of a hack, but it's the easiest way
424    //   to send an email while maintaining flexibility)
425    send_email(
426        app,
427        &email,
428        "Your iGait Submission History",
429        &email_body
430    ).await
431        .context("Failed to send email!")?;
432
433    Ok(())
434}

436/// Generates a PDF file containing the complete history of the user's submissions.
437/// 
438/// # Fails
439/// * If the PDF cannot be generated.
440/// * If the PDF cannot be read.
441/// * If the PDF cannot be uploaded to the user's S3 bucket.
442/// * If the presigned URL cannot be generated.
443/// 
444/// # Panics
445/// * If the PDF cannot be rendered.
446/// 
447/// # Arguments
448/// * `app` - The application state.
449/// * `jobs` - The jobs to include in the PDF.
450/// * `uid` - The user ID.
451/// * `timestamp` - The timestamp to use for the PDF.
452/// 
453/// # Returns
454/// * The email address of the user.
455/// * The presigned URL of the PDF.
456/// 
457/// # Notes
458/// <div class="warning">
459///    This function is currently a bit of a hack. It uses a synchronous closure to generate the PDF.
460///    <br>It is possible for this function to panic without catching the panic.
461///    <br><br>Currently, I do not have the technical skills to fix this. I will come back with more skill later.
462/// </div>
463#[tracing::instrument]
464async fn get_email_and_pdf_link(
465    app: Arc<AppState>,
466    jobs: Vec<(usize, Job)>,
467    uid: String,
468    timestamp: u64
469) -> Result<(String, String)> {
470    let jobs_og = Arc::new(jobs.clone());
471    let uid_og = Arc::new(uid);
472    let timestamp_og = Arc::new(timestamp);
473
474    let jobs_arc = jobs_og.clone();
475    let uid_arc = uid_og.clone();
476    let timestamp_arc = timestamp_og.clone();
477
478    let sync_thread = tokio::task::spawn( async move {
479        let font_family = genpdf::fonts::from_files("pdf_handling/fonts/SourceSansPro", "SourceSansPro", None)
480            .expect("Closure - Failed to load font family!");// Create a document and set the default font family
481        let mut doc = genpdf::Document::new(font_family);
482
483        // Decorate
484        let mut decorator = genpdf::SimplePageDecorator::new();
485        decorator.set_margins(10);
486
487        // Change the default settings
488        doc.set_title("Demo document");
489        doc.set_page_decorator(decorator);
490        
491        // Add the body
492        let title = genpdf::style::StyledString::new("iGait - Complete Historical Results".to_owned(), genpdf::style::Effect::Bold);
493        doc.push(genpdf::elements::Paragraph::new(title));
494
495        // Add each result as its own paragraph
496        for (index, job) in jobs_arc.iter() {
497            let dt_timestamp_utc: DateTime<Utc> = job.timestamp.into();
498            let dt_timestamp_cst = dt_timestamp_utc.with_timezone(&chrono_tz::US::Central);
499            let status = match job.status.code {
500                JobStatusCode::Complete => {
501                    format!(
502                        "Complete - Result: {}", 
503                        job.status.value
504                    )
505                },
506                _ => {
507                    format!(
508                        "{:?} - Additional Information: {}",
509                        job.status.code,
510                        job.status.value
511                    )
512                }
513            };
514
515            // Add basic data
516            doc.push(genpdf::elements::Text::new(genpdf::style::StyledString::new(
517                format!("Submission #{}", index +1),
518                genpdf::style::Style::new()
519            )));
520            doc.push(genpdf::elements::Text::new(genpdf::style::StyledString::new(
521                format!("Submitted on: {}", dt_timestamp_cst.to_string()),
522                genpdf::style::Style::new()
523            )));
524            doc.push(genpdf::elements::Paragraph::new(genpdf::style::StyledString::new(
525                format!("Status: {}", status),
526                genpdf::style::Style::new()
527            )));
528            doc.push(genpdf::elements::Text::new(genpdf::style::StyledString::new(
529                format!("Email: {}", job.email),
530                genpdf::style::Style::new()
531            )));
532
533            // Add patient data
534            doc.push(genpdf::elements::Text::new(genpdf::style::StyledString::new(
535                "Patient Information:".to_owned(),
536                genpdf::style::Style::new()
537            )));
538            doc.push(genpdf::elements::Text::new(genpdf::style::StyledString::new(
539                format!("Age: {}", job.age),
540                genpdf::style::Style::new()
541            )));
542            doc.push(genpdf::elements::Text::new(genpdf::style::StyledString::new(
543                format!("Sex: {}", job.sex),
544                genpdf::style::Style::new()
545            )));
546            doc.push(genpdf::elements::Text::new(genpdf::style::StyledString::new(
547                format!("Height: {}", job.height),
548                genpdf::style::Style::new()
549            )));
550            doc.push(genpdf::elements::Text::new(genpdf::style::StyledString::new(
551                format!("Weight: {}", job.weight),
552                genpdf::style::Style::new()
553            )));
554            doc.push(genpdf::elements::Text::new(genpdf::style::StyledString::new(
555                "",
556                genpdf::style::Style::new()
557            )));
558        }
559        info!("Built PDF file!");
560
561        // Render the file to a compatible Writer
562        let path = format!("pdf_handling/history_requests/{}_{}.html", uid_arc, timestamp_arc);
563        doc.render_to_file(&path)
564            .expect("Closure - Failed to render PDF!");
565
566        info!("Rendered PDF file to {path}");
567    });
568
569    // Await the sync thread
570    sync_thread
571        .await
572        .map_err(|e| anyhow!("{e:#?}"))
573        .context("Failed to generate PDF file!")?;
574    info!("Preparing to upload...");
575
576    // Read the bytes of the file
577    let path = format!("pdf_handling/history_requests/{}_{}.html", uid_og, timestamp_og);
578    info!("Reading file from {path}");
579    let extended_body_byte_vec = tokio::fs::read(&path)
580        .await
581        .context("Failed to read the PDF file!")?;
582
583    // Remove the file
584    tokio::fs::remove_file(&format!("pdf_handling/history_requests/{}_{}.html", uid_og, timestamp_og))
585        .await
586        .context("Failed to remove the PDF file!")?;
587    info!("Removed file!");
588
589    // Put the extended body into the user's S3 bucket
590    info!("Putting file to AWS...");
591    let aws_path = format!("{}/history_requests/{}.pdf", uid_og, timestamp_og);
592    app.bucket
593        .lock().await
594        .put_object(&aws_path, &extended_body_byte_vec)
595        .await 
596        .context("Failed to upload front file to S3! Continuing regardless.")?;
597    info!("Uploaded PDF file to S3!");
598
599    // Generate the presigned URL
600    info!("Generating presigned URL...");
601    let extended_body_url = app
602            .bucket
603            .lock().await
604            .presign_get(aws_path, 86400 * 7, None)
605            .context("Failed to get the front keyframed URL!")?;
606    info!("Generated a presigned URL for the HTML file!");
607    
608    let email = jobs
609        .iter()
610        .next()
611        .ok_or(anyhow!("User has no jobs!"))?
612        .1
613        .email
614        .clone();
615    Ok((email, extended_body_url))
