15/// Sends an email to the specified address with the specified subject and body.
16/// 
17/// # Arguments
18/// * `to` - The email address to send the email to
19/// * `subject` - The subject of the email
20/// * `body` - The body of the email
21/// 
22/// # Fails
23/// * If the `IGAIT_ACCESS_KEY` environment variable is missing
24/// * If the form fails to send to the Cloudflare Worker
25/// 
26/// # Returns
27/// * A successful result if the email was sent
28/// 
29/// # Notes
30/// * The email is sent to the Cloudflare Worker at `https://email-service.igaitniu.workers.dev/`
31/// # The environment variable `IGAIT_ACCESS_KEY` is used to authenticate the request and must be set
32#[tracing::instrument]
33pub async fn send_email (
34    app:     Arc<AppState>,
35    to:      &str,
36    subject: &str,
37    body:    &str
38) -> Result<()> {
39    info!("Sending email to '{to}'...");
40
41    // Post the form to the Cloudflare Worker
42    let destination = Destination::builder()
43        .set_to_addresses(Some(vec![to.into()]))
44        .build();
45    let content = EmailContent::builder()
46        .set_simple(
47            Some(Message::builder()
48                .set_subject(
49                    Some(Content::builder()
50                        .set_data(Some(subject.to_string()))
51                        .build()
52                        .context("Failed to build the subject line for email!")?
53                    )
54                )
55                .set_body(
56                    Some(Body::builder()
57                        .set_html(
58                            Some(Content::builder()
59                                .set_data(Some(body.to_string()))
60                                .build()
61                                .context("Failed to build body for email!")?
62                            )
63                        )
64                        .build()
65                    )
66                )
67                .build())
68        )
69        .build();
70
71    app.aws_ses_client
72        .lock().await
73        .send_email()
74        .from_email_address("noreply@igaitapp.com")
75        .from_email_address_identity_arn("arn:aws:ses:us-east-2:851725269484:identity/noreply@igaitapp.com")
76        .destination(destination)
77        .content(content)
78        .send()
79        .await
80        .context("Failed to send email!")?;
81    info!("Successfully sent email to '{to}'!");
82
83    Ok(())
84}
