1use igait_lib::{StageData, StageStatus, Output};
2use anyhow::{Context, Result};

4async fn detect_cycles (
5    output: &mut Output,
6    logs: &mut String
7) -> Result<StageStatus> {
8    // Check if we're skipping to this stage or later
9    let is_skipping = output.skip_to_stage.map_or(false, |skip| skip >= 5);
10    
11    let pose_estimation_output_dir = &output.canonical_paths.stage_paths.s4_pose_estimation;
12    let pose_estimation_json_dir = pose_estimation_output_dir.join("json");
13    
14    if is_skipping {
15        // When skipping, verify required files exist from stage 4
16        if !tokio::fs::try_exists(&pose_estimation_json_dir).await
17            .context("Failed to check if pose estimation JSON directory exists")? {
18            anyhow::bail!("Skipping to Stage 5 but required files from Stage 4 (Pose Estimation) don't exist at: {}", pose_estimation_json_dir.display());
19        }
20        logs.push_str("Skipping to Stage 5: Verified required files from Stage 4 exist\n");
21    } else {
22        // When not skipping, verify previous stage completed successfully
23        let pose_estimation_stage = output.stages.s4_pose_estimation
24            .as_ref()
25            .context("Stage 4 (Pose Estimation) data missing")?
26            .status
27            .as_ref()
28            .map_err(|e| anyhow::anyhow!("Stage 4 (Pose Estimation) failed: {e}"))?;
29        anyhow::ensure!(matches!(pose_estimation_stage, StageStatus::Done), "Stage 4 (Pose Estimation) did not complete successfully");
30    }
31    
32    Ok(StageStatus::Skipped)
33}
34
35pub async fn execute(
36    output: &mut Output
37) -> StageData {
38    let mut logs = String::new();
39    let status = match detect_cycles(output, &mut logs).await {
40        Ok(status) => Ok(status),
41        Err(e) => {
42            let mut error_body = format!("{e:?}");
43            for (ind, err) in e.chain().enumerate() {
44                error_body.push_str(&format!("{ind}: {err}\n"));
45            }
46            Err(error_body)
47        }
48    };
49    StageData {
50        status,
51        logs,
52    }
53}
