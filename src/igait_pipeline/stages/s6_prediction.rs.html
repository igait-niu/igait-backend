1use std::path::PathBuf;
2
3use igait_lib::{StageData, StageStatus, Output};
4use anyhow::{anyhow, ensure, Context, Result};
5use tokio::process::Command;
6
7// Use local assets directory instead of remote one
8const PATH_TO_STAGE_6_ASSETS: &str = "./igait-pipeline/assets/stage_6";
9
10async fn predict (
11    output: &mut Output,
12    logs: &mut String
13) -> Result<StageStatus> {
14    // Check if we're skipping to this stage or later
15    let is_skipping = output.skip_to_stage.map_or(false, |skip| skip >= 6);
16    
17    let pose_estimation_output_dir = &output.canonical_paths.stage_paths.s4_pose_estimation;
18    let pose_estimation_json_dir = pose_estimation_output_dir.join("json");
19    
20    if is_skipping {
21        // When skipping, verify required files exist from stage 4
22        if !tokio::fs::try_exists(&pose_estimation_json_dir).await
23            .context("Failed to check if pose estimation JSON directory exists")? {
24            anyhow::bail!("Skipping to Stage 6 but required files from Stage 4 (Pose Estimation) don't exist at: {}", pose_estimation_json_dir.display());
25        }
26        logs.push_str("Skipping to Stage 6: Verified required files from Stage 4 exist\n");
27    } else {
28        // When not skipping, verify previous stage completed successfully
29        let pose_estimation_stage = output.stages.s4_pose_estimation
30            .as_ref()
31            .context("Stage 4 (Pose Estimation) data missing")?
32            .status
33            .as_ref()
34            .map_err(|e| anyhow!("Stage 4 (Pose Estimation) failed: {e}"))?;
35        ensure!(matches!(pose_estimation_stage, StageStatus::Done), "Stage 4 (Pose Estimation) did not complete successfully");
36    }
37    
38    let prediction_output_dir = &output.canonical_paths.stage_paths.s6_prediction;
39
40    // Create the output subdirectory if it doesn't exist
41    tokio::fs::create_dir_all(prediction_output_dir).await
42        .context("Failed to create output/s6_prediction directory")?;
43
44    // Copy JSON directory using cp command (since tokio::fs::copy doesn't work for directories)
45    let copy_output = Command::new("cp")
46        .arg("-r")
47        .arg(&pose_estimation_json_dir)
48        .arg(prediction_output_dir)
49        .output()
50        .await
51        .context("Failed to execute cp command for JSON directory")?;
52    ensure!(copy_output.status.success(), "Failed to copy JSON files from Stage 4 (Pose Estimation) to Stage 6 (Prediction)");
53
54    // Build paths for all files necessary for inference
55    let assets_dir_path = &PathBuf::from(PATH_TO_STAGE_6_ASSETS);
56    let model_dir_path = &assets_dir_path.join("Model").canonicalize()
57        .context("Failed to canonicalize Model directory path")?;
58    let python_script_path = &assets_dir_path.join("main.py");
59    let requirements_path = &assets_dir_path.join("requirements.txt");
60
61    // Copy only the Python script and requirements to the output directory
62    let output_python_script = &prediction_output_dir.join("main.py");
63    tokio::fs::copy(python_script_path, output_python_script).await
64        .context("Failed to copy main.py to output/s6_prediction/main.py")?;
65    let output_requirements = &prediction_output_dir.join("requirements.txt");
66    tokio::fs::copy(requirements_path, output_requirements).await
67        .context("Failed to copy requirements.txt to output/s6_prediction/requirements.txt")?;
68
69    // Use bash to run module commands and Python setup
70    // The module command is a bash function, so we need to invoke it through bash -l -c
71    let setup_script = format!(
72        r#"
73        module purge && \
74        module load python/python-3.12.4 && \
75        python3 -m pip install -U pip --user && \
76        pip3 install -r {} --user
77        "#,
78        output_requirements.display()
79    );
80    
81    logs.push_str("Setting up Python environment with modules...\n");
82    let output = Command::new("bash")
83        .arg("-l")
84        .arg("-c")
85        .arg(&setup_script)
86        .output()
87        .await
88        .context("Failed to run Python environment setup")?;
89    
90    let stdout = String::from_utf8_lossy(&output.stdout);
91    let stderr = String::from_utf8_lossy(&output.stderr);
92    logs.push_str(&format!("Setup stdout:\n{stdout}\n"));
93    logs.push_str(&format!("Setup stderr:\n{stderr}\n"));
94    
95    ensure!(output.status.success(), "Python environment setup failed");
96    logs.push_str("Python environment setup complete.\n");
97
98    // Run the Python script with the Model directory path from assets
99    let run_script = format!(
100        r#"
101        module purge && \
102        module load python/python-3.12.4 && \
103        cd {} && \
104        python3 {} {} {} {}
105        "#,
106        prediction_output_dir.display(),
107        output_python_script.display(),
108        prediction_output_dir.display(),
109        prediction_output_dir.display(),
110        model_dir_path.display()
111    );
112    
113    logs.push_str("Running Python inference script...\n");
114    let output = Command::new("bash")
115        .arg("-l")
116        .arg("-c")
117        .arg(&run_script)
118        .output()
119        .await
120        .context("Failed to run Python script")?;
121    
122    let stdout = String::from_utf8_lossy(&output.stdout);
123    let stderr = String::from_utf8_lossy(&output.stderr);
124    logs.push_str(&format!("Python script stdout:\n{stdout}\n"));
125    logs.push_str(&format!("Python script stderr:\n{stderr}\n"));
126    
127    ensure!(output.status.success(), "Running Python script failed");
128    logs.push_str("Python script completed successfully.\n");
129
130    Ok(StageStatus::Done)
131}
132pub async fn execute(
133    output: &mut Output
134) -> StageData {
135    let mut logs = String::new();
136
137    let status = match predict(output, &mut logs).await {
138        Ok(status) => Ok(status),
139        Err(e) => {
140            let mut error_body = format!("{e:?}\n");
141
142            for (ind, err) in e.chain().enumerate() {
143                error_body.push_str(&format!("{ind}: {err}\n"));
144            }
145
146            Err(error_body)
147        }
148    };
149
150    StageData {
151        status,
152        logs,
153    }
154}
